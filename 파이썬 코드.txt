    d = []

def sum_n(n):
    total = 0
    i = 1
    assert i == 1 and total == 0, "i != 1 or total != 0"
    while i <= n:
        assert i <= n and total == i * (i - 1) // 2, 'E1'
        total += i
        assert i <= n and total == i * (i + 1) // 2, 'E2'
        i += 1
    assert i == n + 1 and total == n * (n + 1) // 2, 'E3'
    return total

for n in [10, 20, 100]:
    print(sum_n(n))

def Prim(V, C):
    T = []; U = {1}
    while (U != V):
        small = L
        for u in U:
            for v in V - U:
                if C[u-1][v-1] < small:
                    small = C[u-1][v-1]
                    edge = (u,v)
        T.append(edge)
        U = U | {edge[1]}
    return T

def Kruskal(V, C):
    E = []
    for u in V:
        for v in V:
            if u < v and C[u-1][v-1] < L:
                E.append((u, v, C[u-1][v-1]))
    E.sort(reverse = True, key = lambda x: x[2])
    
    T = []
    M = V[:]
    while len(T) < len(V) - 1:
        u, v, c = E.pop()
        if (M[u-1] == M[v-1]):
            continue
        for w in V:
            if M[w-1] == M[v-1]:
                M[w-1] = M[u-1]
        T.append((u,v))
    return T

def binary_search(x, a, n):
    left = 0; right = n - 1
    while left <= right:
        mid = (left + right) // 2
        print('left/right/mid: %d/%d/%d' % (left, right, mid))
        if a[mid] < x:
            left = mid + 1
        elif a[mid] > x:
            right = mid - 1
        else:
            return mid
    return -1;

def bubble_sort(a, n):

        for j in range(n - i - 1):

                a[j], a[j+1] = a[j+1], a[j]
